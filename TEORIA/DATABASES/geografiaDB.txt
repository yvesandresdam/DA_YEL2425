CREATE TABLE country (
	id SERIAL PRIMARY KEY,
	name VARCHAR(100)
);

CREATE TABLE cstate (
	id INT PRIMARY KEY,
	name VARCHAR(100) NOT NULL,
	id_country INT,
	FOREIGN KEY (id_country) REFERENCES country(id)
);

CREATE TABLE city (
	id INT PRIMARY KEY,
	name VARCHAR(100) NOT NULL,
	id_cstate INT,
	FOREIGN KEY (id_cstate) REFERENCES cstate(id)
);
